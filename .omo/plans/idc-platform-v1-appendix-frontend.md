## D.4 动态路由注册机制

```typescript
// router/dynamic-routes.ts
// 登录后根据后端返回的菜单配置动态注册路由

interface MenuItem {
  path: string
  name: string
  component: string  // 组件路径，如 "admin/order/OrderList.vue"
  meta: {
    title: string
    icon: string
    permissions?: string[]
    hidden?: boolean
  }
  children?: MenuItem[]
}

// 登录后调用
async function initDynamicRoutes(role: string) {
  const menuTree = await fetchMenuTree(role)  // GET /api/v1/auth/menus
  
  // 递归注册路由
  function registerRoutes(menus: MenuItem[], parentRoute?: string) {
    menus.forEach(menu => {
      const route: RouteRecordRaw = {
        path: menu.path,
        name: menu.name,
        component: () => import(`../views/${menu.component}.vue`),
        meta: menu.meta
      }
      
      if (menu.children?.length) {
        router.addRoute(parentRoute || 'layout', route)
        registerRoutes(menu.children, menu.name)
      } else {
        if (parentRoute) {
          router.addRoute(parentRoute, route)
        } else {
          router.addRoute(route)
        }
      }
    })
  }
  
  registerRoutes(menuTree)
}
```
